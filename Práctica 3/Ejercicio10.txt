process personas[id: 0..N-1]{
    Juego.llegar();
    Usar_Juego();
    Juego.devolverJuego();
}

process empleado{
    for int i = 1..N{
        Desinfectar_Juego();
        Juego.avisar();
        Juego.esperarJuego();
    }
}

monitor Juego{
    cond jugar,empl,usado;
    int esperando = 0;

    Procedure llegar(){
        esperando++;
        signal(empl);
        wait(jugar);
    }

    Procedure avisar(){
        if (esperando == 0)
            wait(empl);
        esperando--;
        signal(jugar);
    }

    Procedure devolverJuego(){
        signal(usado);
    }

    Procedure esperarJuego(){
        wait(usado);
    }
}
