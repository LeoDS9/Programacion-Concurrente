process Persona[id: 0..N-1]{
    Solicitud s;
    Resultado res;
    Oficina.llegar(id,s);
    Oficina.obtenerResultado(id,res);
}

process empleado{
    Solicitud s;
    int idP;
    Resultado res;
    for int i = 1..N{
        Oficina.esperarPersona();
        Oficina.obtenerSolicitud(idP,s);
        res = resolverTramite(s);
        Oficina.entregarResultado(idP,res);
    }
}


monitor Oficina{
    Cola cola;
    cond espera[N], esperaEmp;
    int esperando = 0;
    Resultado res[N];

    Procedure llegar(id: in int, s: in Solicitud, r: out Resultado){
        Push(cola,(id,s));
        esperando++;
        signal(esperaEmp);
        wait(espera[id]);
        r = res[id];
    }

    Procedure esperarPersona(leer){
        if (esperando == 0){
            Delay(10);
            wait(esperaEmp);
        }
    }

    Procedure obtenerSolicitud(idP: out int, s: out Solicitud){
        Pop(cola, (idP,s));
        esperando--;
    }

    Procedure entregarResultado(idP: in int, r: in Resultado){
        res[idP] = r;
        signal(espera[idP]);
    }
    
}
